# Ejecutar en una celda de cÃ³digo para generar TOC automÃ¡ticamente
import re

def generate_toc_from_notebook():
    """
    Genera una tabla de contenido automÃ¡tica basada en los encabezados Markdown
    del notebook actual. Para usar en Google Colab.
    """
    # SimulaciÃ³n de encabezados encontrados en el notebook
    headers = [
        "# Proyecto de AnÃ¡lisis de Datos",
        "## 1. IntroducciÃ³n",
        "### 1.1 Objetivos",
        "### 1.2 MetodologÃ­a", 
        "## 2. Carga y ExploraciÃ³n de Datos",
        "### 2.1 ImportaciÃ³n de LibrerÃ­as",
        "### 2.2 Carga del Dataset",
        "## 3. AnÃ¡lisis Exploratorio",
        "### 3.1 EstadÃ­sticas Descriptivas",
        "### 3.2 Visualizaciones",
        "## 4. Preprocesamiento",
        "## 5. Modelado",
        "### 5.1 SelecciÃ³n del Modelo",
        "### 5.2 Entrenamiento",
        "## 6. EvaluaciÃ³n",
        "## 7. Conclusiones"
    ]
    
    toc_lines = ["## ğŸ“‹ Tabla de Contenido\n"]
    
    for header in headers:
        # Extraer nivel y texto del encabezado
        level = len(re.match(r'^#+', header).group())
        text = header.lstrip('#').strip()
        
        # Crear ancla (link interno)
        anchor = text.lower().replace(' ', '-').replace('.', '')
        anchor = re.sub(r'[^\w\-]', '', anchor)
        
        # Generar indentaciÃ³n segÃºn el nivel
        indent = '  ' * (level - 1)
        
        # Agregar nÃºmero si no lo tiene
        if level == 2 and not re.match(r'^\d+\.', text):
            continue  # Saltar el tÃ­tulo principal
        
        toc_lines.append(f"{indent}- [{text}](#{anchor})")
    
    return '\n'.join(toc_lines)

# Generar y mostrar la TOC
toc_markdown = generate_toc_from_notebook()
print("Copia este cÃ³digo en una celda Markdown:")
